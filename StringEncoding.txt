public class Main
{
	public static void main(String[] args) {
		String inputString="Gnana Moorthy";
		for(int i=0;i<inputString.length();i++){
		    char eachCharacter=inputString.charAt(i);
		    int ascii=(int) eachCharacter;
		    
		    if(eachCharacter>='a' && eachCharacter<='z'){
		        ascii=ascii-(int)'a'+1;
		    }else if(eachCharacter>='A' && eachCharacter<='Z'){
		        ascii=ascii-(int)'A'+27;
		    }else if(eachCharacter=='.'){
		        ascii=99;
		    }else if(eachCharacter==' '){
		        ascii=0;
		    }else{
		        ascii=90;
		    }
		    
		    int lengthOfStringWithIncrement=inputString.length()*100;
		    if(eachCharacter>='A' && eachCharacter<='z'){
		        System.out.println(ascii+lengthOfStringWithIncrement);
		    }else{
		        System.out.println(ascii);
		    }
		    
		    
		}
		System.out.println("Thank Your This Is My Solution For String Encryption");
	}
}
